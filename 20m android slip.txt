import androidx.annotation.NonNull;

import android.os.Bundle;

import android.view.ViewGroup;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;
public class MainActivity extends AppCompatActivity {
    Button submit;
    EditText name,email,password,number,age;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        name = findViewById(R.id.name);
        email = findViewById(R.id.email);
        password = findViewById(R.id.password);
        number = findViewById(R.id.number);
        age = findViewById(R.id.age);
        submit = findViewById(R.id.submit);
        submit.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String nme = name.getText().toString().trim();
                String mail = email.getText().toString().trim();
                String pass = password.getText().toString().trim();
                String num = number.getText().toString().trim();
                String age =  number.getText().toString().trim();

                if(nme.isEmpty() || mail.isEmpty() || pass.isEmpty() || num.isEmpty() || age.isEmpty()){
                    displayMessage("All fields are required.");
                }
                else{
                    Toast.makeText(MainActivity.this, "Data Entered Successfully", Toast.LENGTH_SHORT).show();
                }
            }
        });
    }

    private void displayMessage(String message) {
        MessageDialogFragment dialog = new MessageDialogFragment(message);
        dialog.show(getSupportFragmentManager(), "MessageDialogFragment");
    }


}

package com.example.spinner;
import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Spinner;
import android.widget.TextView;
import android.widget.Toast;
import java.util.ArrayList;
import java.util.List;
public class MainActivity extends AppCompatActivity {
    Button Add,Remove;
    EditText value;
    Spinner spinner;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        Add = findViewById(R.id.Add);
        Remove = findViewById(R.id.remove);
        value = findViewById(R.id.value);
        spinner = findViewById(R.id.spinner);
        List values = new ArrayList();
        Add.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String inputValue = value.getText().toString().trim();
                if (!inputValue.isEmpty()) {
                    values.add(inputValue);
                    ArrayAdapter dataAdapter = new ArrayAdapter(MainActivity.this, android.R.layout.simple_spinner_item, values);
                    dataAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
                    spinner.setAdapter(dataAdapter);
                    Toast.makeText(MainActivity.this, "Added Successfully", Toast.LENGTH_SHORT).show();
                }
                else{
                    Toast.makeText(MainActivity.this, "Enter a value", Toast.LENGTH_SHORT).show();
                }
            }
        });
        Remove.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String inputValue = value.getText().toString().trim();
                if (values.contains(inputValue)) {
                    values.remove(inputValue);
                    ArrayAdapter dataAdapter = new ArrayAdapter(MainActivity.this, android.R.layout.simple_spinner_item, values);
                    dataAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
                    spinner.setAdapter(dataAdapter);
                    Toast.makeText(MainActivity.this, "Removed Successfully", Toast.LENGTH_SHORT).show();
                }
                else{
                    Toast.makeText(MainActivity.this, "Value not Present", Toast.LENGTH_SHORT).show();
                }
            }
        });
    }

}

        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        list = findViewById(R.id.l1);
        add = findViewById(R.id.add);
        remove = findViewById(R.id.remove);
        filename= findViewById(R.id.fn);
        search = findViewById(R.id.search);
        fileExplorer.add("Data");
        fileExplorer.add("Android");
        fileExplorer.add("DCMI");
        fileExplorer.add("Download");
        ArrayAdapter  adapter = new ArrayAdapter(MainActivity.this, androidx.appcompat.R.layout.support_simple_spinner_dropdown_item,fileExplorer);
        list.setAdapter(adapter);
        add.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String name = filename.getText().toString().trim() ;
                fileExplorer.add(name);
                ArrayAdapter adapter = new ArrayAdapter(MainActivity.this, com.google.android.material.R.layout.support_simple_spinner_dropdown_item,fileExplorer);
                list.setAdapter(adapter);
            }
        });
        remove.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String name = filename.getText().toString().trim() ;
                fileExplorer.remove(name);
                ArrayAdapter adapter = new ArrayAdapter(MainActivity.this, com.google.android.material.R.layout.support_simple_spinner_dropdown_item,fileExplorer);
                list.setAdapter(adapter);
            }
        });
        search.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String name = filename.getText().toString().trim() ;
                if(fileExplorer.contains(name)){
                    int selectedId = list.getSelectedItemPosition();
                    Toast.makeText(MainActivity.this, "File present at "+selectedId, Toast.LENGTH_SHORT).show();
                }
            }
        });
    }
